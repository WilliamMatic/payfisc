1. Récupérer les Informations d'une Déclaration

GET /api/paiement/declaration-info.php?reference=DEC202412011430123456

Paramètres :

reference: "DEC202412011430123456" // (requis) Référence de la déclaration
JavaScript
PHP
async function getDeclarationInfo(reference) {
  const response = await fetch(
    `https://api.mpako.net/api/paiement/declaration-info.php?reference=${encodeURIComponent(reference)}`, 
    {
      method: 'GET',
      headers: headers
    }
  );
  
  const result = await response.json();
  return result;
}

// Utilisation
getDeclarationInfo('DEC202412011430123456').then(result => {
  if (result.status === 'success') {
    console.log('Montant total:', result.data.declaration.montant_total);
    console.log('Pénalité:', result.data.declaration.penalite);
  } else {
    console.error('Erreur:', result.message);
  }
});
function getDeclarationInfo(reference) {
  $url = 'https://api.mpako.net/api/paiement/declaration-info.php?reference=' . urlencode(reference);
  
  $ch = curl_init();
  curl_setopt_array($ch, [
    CURLOPT_URL => $url,
    CURLOPT_HTTPHEADER => $headers,
    CURLOPT_RETURNTRANSFER => true
  ]);
  
  $response = curl_exec($ch);
  $result = json_decode($response, true);
  curl_close($ch);
  
  return $result;
}

// Utilisation
$result = getDeclarationInfo('DEC202412011430123456');
if ($result['status'] === 'success') {
  echo "Montant: " . $result['data']['declaration']['montant_total'];
} else {
  echo "Erreur: " . $result['message'];
}

2. Traiter un Paiement

POST /api/paiement/traiter-paiement.php

Body :

{
  "reference_declaration": "DEC202412011430123456",
  "methode_paiement": "mobile_money"
}
JavaScript
async function traiterPaiement(referenceDeclaration, methodePaiement) {
  const response = await fetch('https://api.mpako.net/api/paiement/traiter-paiement.php', {
    method: 'POST',
    headers: headers,
    body: JSON.stringify({
      reference_declaration: referenceDeclaration,
      methode_paiement: methodePaiement
    })
  });
  
  return await response.json();
}

// Utilisation
traiterPaiement('DEC202412011430123456', 'mobile_money')
  .then(result => {
    if (result.status === 'success') {
      console.log('Paiement traité!');
      console.log('Référence bancaire:', result.data.reference_bancaire);
    }
  });

class DeclarationPaymentClient {
  constructor(apiKey, bankId, baseUrl = 'https://api.mpako.net/api/paiement') {
    this.apiKey = apiKey;
    this.bankId = bankId;
    this.baseUrl = baseUrl;
    this.headers = {
      'X-API-Key': apiKey,
      'X-Bank-ID': bankId,
      'Content-Type': 'application/json'
    };
  }

  // Récupérer les infos d'une déclaration
  async getDeclarationInfo(reference) {
    const response = await fetch(
      `${this.baseUrl}/declaration-info.php?reference=${encodeURIComponent(reference)}`, 
      {
        method: 'GET',
        headers: this.headers
      }
    );
    
    return await response.json();
  }

  // Traiter un paiement
  async traiterPaiement(referenceDeclaration, methodePaiement) {
    const response = await fetch(`${this.baseUrl}/traiter-paiement.php`, {
      method: 'POST',
      headers: this.headers,
      body: JSON.stringify({
        reference_declaration: referenceDeclaration,
        methode_paiement: methodePaiement
      })
    });
    
    return await response.json();
  }

  // Annuler une déclaration
  async annulerDeclaration(referenceDeclaration) {
    const response = await fetch(`${this.baseUrl}/annuler-declaration.php`, {
      method: 'POST',
      headers: this.headers,
      body: JSON.stringify({
        reference_declaration: referenceDeclaration
      })
    });
    
    return await response.json();
  }
}

// Utilisation
const client = new DeclarationPaymentClient('votre_cle_api', 'votre_bank_id');

// Exemple d'utilisation complète
async function processDeclarationComplete() {
  // 1. Récupérer les infos de la déclaration
  const declarationInfo = await client.getDeclarationInfo('DEC202412011430123456');
  
  if (declarationInfo.status === 'success') {
    const data = declarationInfo.data;
    
    // Afficher les informations
    console.log('Contribuable:', data.contribuable.raison_sociale);
    console.log('Montant total:', data.declaration.montant_total);
    
    // 2. Traiter le paiement si l'utilisateur confirme
    const paiementResult = await client.traiterPaiement(
      'DEC202412011430123456', 
      'mobile_money'
    );
    
    if (paiementResult.status === 'success') {
      console.log('Paiement initié avec référence:', paiementResult.data.reference_bancaire);
    }
  }
}
